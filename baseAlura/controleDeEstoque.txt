using System.Reflection;

Dictionary<string, int> itens = new Dictionary<string, int>();


ExibirMenu();

void ExibirMenu()
{
    Console.Clear();
    Console.WriteLine(@"
    Digite 1 para adicionar um item ao Estoque.
    Digite 2 para alterar estoque de um item.
    Digite 3 para ver o estoque de um item.
    Digite 0 para sair.
    ");

    Console.Write("\nDigite a opção que você deseja escolher: ");
    int opcao = int.Parse(Console.ReadLine()!);

    Thread.Sleep(1000);
    Console.Clear();
   
    switch(opcao)
    {
        case 1:AdicionarItem();
        break;
        case 2: AlterarEstoque();
        break;
        case 3: ExibirEstoque();
        break;
        case 0: Console.WriteLine("Obrigado por usar o nosso sistema!!");
        break;
    }


}

void AdicionarItem()
{
    Console.Write("Digite o nome do item que você quer adicionar ao estoque: ");
    string item = Console.ReadLine()!;

    itens.Add(item, 0);

    Console.WriteLine($"\n{item} adicionado ao estoque");
    Console.Write("Digite qualquer tecla para voltar ao menu: ");
    Console.ReadKey();

    ExibirMenu();
}

void AlterarEstoque()
{
    Console.Write("Digite o nome do item que você deseja alterar o estoque: ");
    string item = Console.ReadLine()!;

    if(itens.ContainsKey(item))
    {
        Console.WriteLine($"\nDigite a quantidade de estoque que você deseja adicionar ao {item}: ");
        int quantidade = int.Parse(Console.ReadLine()!);

        itens[item] = (quantidade);

        Thread.Sleep(1500);
        Console.WriteLine($"Agora o {item} tem estoque de {quantidade} unidades");
     Console.Write("Digite qualquer tecla para voltar ao menu: ");
    Console.ReadKey();

    ExibirMenu();
    }
    else
    {
        Console.WriteLine($"\nNão temos {item} em nossa loja ");
        Console.Write("Digite qualquer tecla para voltar ao menu:");
        Console.ReadKey();
        ExibirMenu();


    }
}


void ExibirEstoque()
{
    Console.Write("Digite o nome do item que você deseja ver o estoque: ");
    string item = Console.ReadLine()!;

     if(itens.ContainsKey(item))
    {
       int quantidade = itens[item];
       Console.WriteLine($"A quantidade de {item} que temos no nosso estoque é {quantidade}");
     Console.Write("Digite qualquer tecla para voltar ao menu: ");
    Console.ReadKey();

    ExibirMenu();
    }
    else
    {
        Console.WriteLine($"\nNão temos {item} em nossa loja ");
        Console.Write("Digite qualquer tecla para voltar ao menu:");
        Console.ReadKey();
        ExibirMenu();


    }
}
