static ContaCorrente ConverterStringParaContaCorrente(string linha)
{
    // 375 4644 2483.13 Jonatan

    var campos = linha.Split(',');
    var agencia = int.Parse(campos[0]);
    var numero = int.Parse(campos[1]);
    var saldoComDouble = double.Parse(campos[2].Replace('.',','));
    var titularNome = campos[3];

    Cliente titular = new();
    titular.Nome = titularNome;

    var resultado = new ContaCorrente(agencia, numero);
    resultado.Depositar(saldoComDouble);
    resultado.Titular = titular;

    return resultado;
}

static void CriarArquivo()
{
    
        var caminhoDoArquivo = "contasExportadas.csv";

        using(var fluxoDoArquivo = new FileStream( caminhoDoArquivo, FileMode.Create))
        {
            var contaComoString = "456,7895,4789.48,Arthur Rafael";

            var encoding = Encoding.UTF8;
            var bytes = encoding.GetBytes(contaComoString);

            fluxoDoArquivo.Write(bytes,0,bytes.Length);
            
        }
    
}

    static void UsarStreamDeEntrada()
    {
        using (var fluxoDeEntrada = Console.OpenStandardInput())
        using (var fs = new FileStream("entradaConsole.txt", FileMode.Create))
        {
            var buffer = new byte[1024]; //1kb

            while(true)
            {
                var byteslidos = fluxoDeEntrada.Read(buffer, 0, 1024);

                fs.Write(buffer, 0, byteslidos);
                fs.Flush();

                Console.WriteLine($"Bytes lidos na console: {byteslidos}");
            }
        }
