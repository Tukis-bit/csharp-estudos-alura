namespace Alura.Filmes;

class Filmes
{
    private List<string> elenco = new();
    public Filmes(string titulo)
    {
        Titulo = titulo;
    }

    public string Titulo { get; }
    public int Duracao { get; set; }
    
    public void AdicionarParticipante(string participante)
    {
        elenco.Add(participante);
    }

    public void MostrarElenco()
    {
        Console.WriteLine($"Elenco do filme {Titulo}:\n");
        elenco.ForEach( e => Console.WriteLine($"Participante: {e}"));
        
    } 
}
